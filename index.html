<script>  
/* YOUR REAL PHOTOS */  
const photos = [  
"photo1.jpg",  
"photo2.jpg",  
"photo3.jpg",  
"photo4.jpg"  
];  
let slideIndex=0;  
setInterval(()=>{  
slideIndex=(slideIndex+1)%photos.length;  
document.getElementById("slideImg").src=photos[slideIndex];  
},2500);  
/* QUIZ */  
const questions = [  
{ q:"Who loves you the most?", options:["Mansi ","Only Mansi ","Obviously Mansi "] },  
{ q:"Who makes your life magical?", options:["Mansi","Mansi Forever","My Wife Mansi "] },  
{ q:"Who is your favourite person?", options:["Mansi ","Love of my life ","My always "] },  
{ q:"Ready for the most important question?", options:["Yes","Of course","Always ready"] },  
{ q:"Divesh will you be my Valentine?", options:["Yes ","No "] }  
];  
let index=0;  
const qEl=document.getElementById("question");  
const optEl=document.getElementById("options");  
function loadQuestion(){  
qEl.innerText=questions[index].q;  
optEl.innerHTML="";  
questions[index].options.forEach(opt=>{  
const btn=document.createElement("button");  
btn.innerText=opt;  
if(opt.includes("No")){  
btn.id="noBtn";  
btn.onmouseover=runAway;  
btn.ontouchstart=runAway;  
}else{  
btn.onclick=next;  
}
optEl.appendChild(btn);  
});  
}
function runAway(){  
const btn=document.getElementById("noBtn");  
const x=Math.random()*250-125;  
const y=Math.random()*150-75;  
btn.style.transform=`translate(${x}px,${y}px)`;  
}
function next(){  
index++;  
if(index<questions.length){  
loadQuestion();  
}else{  
showLove();  
}
}
function showLove(){  
document.getElementById("quizBox").classList.add("hidden");  
document.getElementById("finalBox").classList.remove("hidden");  
for(let i=0;i<70;i++){  
const heart=document.createElement("div");  
heart.className="heart";  
heart.innerText=" ";  
heart.style.left=Math.random()*100+"vw";  
heart.style.animationDuration=(Math.random()*2+3)+"s";  
document.body.appendChild(heart);  
}
}
loadQuestion();
